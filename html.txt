<p>this is my method:</p>

<pre><code>Meteor.methods({
    'verifyUserOwnership': function() {
        function checkCard() {
            return false; // for testing, simulating the bad answer from api
        }

        var checkOwnership = function(nbrTry, delay){
            if (checkCard()) {
              //I will do something if API responds with wwhat I want 
            } else {
                if(nbrTry &lt; 3){
                    delay +=5000;
                    Meteor.setTimeout(function(){checkOwnership(nbrTry+1, delay);}, delay);
                } else {
                    throw new Meteor.Error('Card could not be found' );// has to be sent to client if after all the tries, API did respond with somwthing bad
                }
            }
        }
        checkOwnership(0, 0);
    }
});

    <div class="post-text" itemprop="text">
<p>Use node fibers to stop/restrict recursive callbacks until they return a value. So you mimic blocking.</p>

<p>First in your meteor app dir install: meteor npm install --save fibers</p>

<p>then change your code like so:</p>

<pre><code>if (Meteor.isServer) {
var Future = Npm.require('fibers/future');


Meteor.methods({
    'verifyUserOwnership': function() {
        var fut = new Future();
        function checkCard() {
            return false; // for testing, simulating the bad answer from api
        }

        var checkOwnership = function(nbrTry, delay){
            if (checkCard()) {
              //I will do something if API responds with wwhat I want 
            } else {
                if(nbrTry &lt; 3){
                    delay +=5000;
                    nbrTry +=1;
                    console.log(nbrTry)
                    console.log("out")
                    Meteor.setTimeout(function(){
                        console.log("in")
                        checkOwnership(nbrTry, delay);
                        fut.return('done');
                        return 'done';
                    }, delay);
                } else {
                    console.log('err');
                    throw new Meteor.Error('Card could not be found' );// has to be sent to client if after all the tries, API did respond with somwthing bad
                }
            }
        }
        if (nbrTry = 0)
            console.log("call")
            checkOwnership(0,0);
    }
});
}
</code></pre>

<p>and notice that your code now has to run server side. Check logs server side!</p>

<p>This code does return the error: Error: Future resolved more than once</p>

<p>but the final error returned in the chain is what you want as per your conditions:[Card could not be found]</p>

<p>Also see <a href="http://stackoverflow.com/questions/28069273/why-will-my-subsequent-meteor-method-calls-not-wait-for-the-first-one-to-finish">this link</a></p>
